\chapter{Program overview}

This is a blueprint for a \emph{proof theory / ordinal analysis of Peano Arithmetic, carried out in
Lean~4}.  The summit of the program is the \textbf{Kirby--Paris independence theorem}: Peano
Arithmetic does not prove that every Goodstein sequence terminates
(\(\mathsf{PA} \nvdash \gamma\)).  As of this writing that summit has \emph{never} been formalized in
\emph{any} proof assistant, by any route --- only the \emph{termination} of Goodstein sequences is
formalized (Coq \texttt{hydra-battles}, Isabelle/AFP \texttt{Goodstein\_Lambda}, and Lean
\texttt{mathlib}/\texttt{lean-gallery}), and abstract Gödel incompleteness is a separate lineage
(Foundation, Paulson, O'Connor).  A Lean proof of the independence is a first.

Everything below is organized as a mountain.  Chapter~\ref{chp:shores} (\emph{Two shores}) records
the two bodies of \emph{already-formalized} machinery this program stands between: mathlib's ordinal
apparatus and Foundation's first-order + incompleteness apparatus.  Chapter~\ref{chp:foothills}
(\emph{Foothills}) is the reusable toolkit --- the faithful encoding and faithfulness bridge (done),
plus the deep gaps (Hardy/Wainer classification, arithmetic transfinite induction).
Chapter~\ref{chp:routeB} is the \textbf{mainline} route (Wainer growth rates); Chapter~\ref{chp:routeA}
is the \textbf{banked} route (internalized Gentzen cut-elimination), which carries the preserved
crux-2 milestone chain.  Chapter~\ref{chp:crown} is the crown: the one ordinal-analysis monument both
routes ultimately require, and the sibling summits it unlocks.

The single, intrinsic obstruction shared by every route is the \emph{ordinal analysis of PA} --- the
\(\varepsilon_0\)-bounded (infinitary) cut-elimination that yields \(\mathrm{TI}(\varepsilon_0)\vdash
\mathrm{Con}(\mathsf{PA})\) (Route~A face) or, equivalently, the Buchholz--Wainer bounding theorem
(Route~B face).  It exists in no Lean library and must be \emph{originated}; the only precedent is the
Coq development of Bryce \& Goré (\texttt{arXiv:2603.00487}).  The lower-bound half of the climb
(\(\text{goodsteinLength}\) dominates every fixed fast-growing level below \(\varepsilon_0\)) is
already done and axiom-clean.

\section{mathlib: ordinal notation and the fast-growing hierarchy}

\begin{definition}\label{def:onote}
  \lean{ONote}
  \leanok
mathlib's constructive ordinal notation system \texttt{ONote} (Cantor normal form as an inductive
type), together with its normal-form predicate \texttt{ONote.NF}.  This is the concrete, computable
carrier for every ordinal \(<\varepsilon_0\) used throughout the program.  (mathlib also has the
semantic \texttt{Ordinal.CNF} decomposition, but it is not in this build's import closure; the notation
system \texttt{ONote} is what the growth theory runs on.)
\end{definition}

\begin{definition}\label{def:veblen}
  \lean{Ordinal.veblen}
  \leanok
The two-argument Veblen function \(\varphi\), from which mathlib defines the \(\varepsilon\)-function
\(\varepsilon_o = \texttt{veblen}\,1\,o\) and hence the ceiling \(\varepsilon_0\).  This is the
semantic ordinal boundary that PA's proof-theoretic ordinal must equal; \texttt{ONote} notations are
its below-\(\varepsilon_0\) codes.
\end{definition}

\begin{definition}\label{def:fastgrowing}
  \lean{ONote.fastGrowing, ONote.fundamentalSequence}
  \uses{def:onote}
  \leanok
The fast-growing hierarchy \(f_o : \mathbb{N}\to\mathbb{N}\) on ordinal notations
(\(f_0 = \mathrm{succ}\), \(f_{o+1}(n) = f_o^{[n]}(n)\), \(f_\lambda(n) = f_{\lambda[n]}(n)\) at limits
via \texttt{fundamentalSequence}).  mathlib supplies the definition and small values but \emph{none} of
the growth theory (expansiveness, monotonicity, domination); those are developed in this program's
Hardy port and Domination modules.
\end{definition}

\section{The Goodstein object and its faithful encoding}

\begin{theorem}\label{thm:goodstein_termination}
  \lean{GoodsteinPA.Dom.goodstein_terminates}
  \leanok
\textbf{Goodstein termination (the mathematical fact).}  Every Goodstein sequence reaches \(0\):
\(\forall m,\ \exists N,\ \texttt{goodsteinSeq}\,m\,N = 0\).  Proved in-repo
(\texttt{Dom.goodstein\_terminates}, the \texttt{lean-gallery} artifact's proof; mathlib carries the
ingredients via \texttt{ONote.fastGrowing} but upstreaming this theorem is a non-goal of this program).
It plays two roles here: the \emph{faithfulness target} the encoding is checked against
(Thm~\ref{thm:bridge}), and the totality witness behind the length function
(Def~\ref{def:goodstein_length}).  A green anchor, not an obligation.
\end{theorem}

\end{definition}

\begin{definition}\label{def:wainer_class}
  \uses{def:hardy, def:fastgrowing, def:veblen, thm:hauptsatz}
  \notready
\textbf{Wainer/Buchholz classification (the deep gap = ordinal analysis of PA).}  A recursive function
is \(\mathsf{PA}\)-provably total iff it is eventually dominated by some \(f_o\) with \(o<\varepsilon_0\).
Its \emph{converse} half (provably-total \(\Rightarrow\) \(<\varepsilon_0\)-bounded) is proved in the
literature by embedding PA into \(\omega\)-logic, applying cut-elimination, and reading off an ordinal
assignment \(<\varepsilon_0\) (Buchholz--Wainer 1987).  Formalized in no proof assistant; this is the
Route-B face of the one intrinsic monument.
\end{definition}

\begin{definition}\label{def:encoding}
  \lean{GoodsteinPA.goodsteinSentence}
  \uses{thm:goodstein_termination}
  \leanok
The first-order \textbf{Goodstein sentence} \(\gamma = \texttt{goodsteinSentence}\), the faithful
\(\mathcal{L}_{\mathrm{OR}}\)-encoding of ``every Goodstein sequence terminates''.  This is the object of
the entire independence claim; if it is unfaithful, the whole result is worthless --- hence the bridge
below.
\end{definition}

\begin{theorem}\label{thm:bridge}
  \lean{GoodsteinPA.goodsteinSentence_faithful}
  \uses{def:encoding, thm:goodstein_termination}
  \leanok
\textbf{Faithfulness bridge (locked anchor).}  The standard model satisfies the encoded sentence iff
every genuine hereditary-base Goodstein sequence reaches \(0\):
\(\;(\mathbb{N}\models\gamma)\iff\forall m,\exists N,\ \texttt{goodsteinSeq}\,m\,N = 0\).  Proved and
kernel-clean (\(\{\texttt{propext},\texttt{Classical.choice},\texttt{Quot.sound}\}\)).  This is what
makes a \(\mathsf{PA}\nvdash\gamma\) result about the \emph{real} Goodstein theorem.
\end{theorem}


\chapter{Route B --- Wainer growth rates (mainline)}\label{chp:routeB}

The route adopted after the 2026-07-01 route decision.  It proves non-provability from \emph{rates of
growth} and, unlike Route~A, needs no internalized cut-elimination and no Gödel-II wiring: if
\(\mathsf{PA}\) proved \(\gamma\), Wainer would bound the Goodstein length function below one fixed
\(f_o\) with \(o<\varepsilon_0\); but the Cichoń/Caicedo lower bound (already machine-checked) shows no
fixed \(f_o\) can bound it.  The lower-bound half is done and clean; the single remaining gap is the
Wainer classification (Def~\ref{def:wainer_class}), surfaced here as one named axiom.

\begin{theorem}\label{thm:dom_lower}
  \lean{GoodsteinPA.Dom.goodsteinLength_dominates_fastGrowing}

  \leanok
\textbf{Cichoń/Caicedo lower bound (done).}  For every normal-form \(o\), the Goodstein length function

\end{theorem}

\begin{lemma}\label{lem:succ_gap}
  \lean{GoodsteinPA.WainerRoute.fastGrowing_fixed_add_two_lt_successor, GoodsteinPA.WainerRoute.add_le_iterate_of_lt}
  \uses{def:fastgrowing}
  \leanok

\(f_o(m) + 2 < f_{o+1}(m)\), via the elementary engine ``iterating a strictly inflationary map adds at
least the iteration count'' (\texttt{add\_le\_iterate\_of\_lt}).  This is what lets the lower bound's
additive \(+2\) slack be swallowed at the notation-successor level.
\end{lemma}

\begin{theorem}\label{thm:no_fixed_bound}
  \lean{GoodsteinPA.WainerRoute.cichon_caicedo_not_eventually_bounded_by_fixed_fastGrowing, GoodsteinPA.WainerRoute.goodsteinLength_eventually_strictly_dominates_fixed_fastGrowing}

exact contradiction target the Wainer bound must collide with; carries the same finite
\texttt{native\_decide} base cases as Thm~\ref{thm:dom_lower}, no debt axiom.
\end{theorem}
